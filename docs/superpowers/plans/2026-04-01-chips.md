# Chips — Single Option Selection in Chat — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Show clickable chip buttons above the chat text input when the AI asks an enum-field question, so the user can tap a chip to reply instantly.

**Architecture:** The backend detects which enum field the AI asked about by keyword-matching the completed message text, then attaches `options: [{label, value}]` to that message via `partialUpdateMessage`. The frontend `Composer` reads the last message's `options` field and renders a `ChipBar` component above the text input. Clicking a chip sends the label as a user message.

**Tech Stack:** React 19, TypeScript, stream-chat-react, Anthropic SDK, Vitest

export type ChipOption = { label: string; value: string };

function camelToTitle(camel: string): string {
  return camel
    .replace(/([A-Z])/g, ' $1')
    .replace(/^./, (c) => c.toUpperCase())
    .trim();
}

export function getEnumOptions(enumValues: string[]): ChipOption[] {
  return enumValues.map((value) => {
    const lastSegment = value.split('.').at(-1) ?? value;
    return { label: camelToTitle(lastSegment), value };
  });
}
```

import { getEnumOptions, type ChipOption } from './getEnumOptions';

// Ordered: father/mother must come before breed to avoid substring false-match on "breed"
const FIELD_KEYWORDS: [fieldName: string, keywords: string[]][] = [
  ['father_breed', ['father']],
  ['mother_breed', ['mother']],
  ['breed',        ['breed']],
  ['advert_type',  ['type of advert', 'advert type', 'what type', 'type would']],
];

export function detectEnumOptions(
  text: string,
  schema: Record<string, { enum?: string[] }>,
): ChipOption[] | null {
  const lower = text.toLowerCase();
  for (const [fieldName, keywords] of FIELD_KEYWORDS) {
    const fieldDef = schema[fieldName];
    if (!fieldDef?.enum) continue;
    if (keywords.some((k) => lower.includes(k))) {
      return getEnumOptions(fieldDef.enum);
    }
  }
  return null;
}
```
